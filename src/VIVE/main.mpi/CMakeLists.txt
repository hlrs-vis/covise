USING(MPI)
USING(vsg)


PROJECT(VIVE.mpi CXX C)
ADD_DEFINITIONS(-DMPI_VIVE)

SET(SOURCES
  ../main/VIVEMain.cpp
)

ADD_COVISE_MODULE(Renderer VIVE.mpi)
COVISE_ADJUST_OUTPUT_DIR(VIVE.mpi "" bin)
target_include_directories(VIVE.mpi SYSTEM PUBLIC ${PTHREADS_INCLUDE_DIR})

TARGET_LINK_LIBRARIES(VIVE.mpi  vvCore
      ${COVISE_VRBCLIENT_LIBRARY} ${COVISE_GRMSG_LIBRARY}
      ${COVISE_UTIL_LIBRARY} ${COVISE_NET_LIBRARY}
      ${EXTRA_LIBS} ${CMAKE_THREAD_LIBS_INIT})
IF(COVISE_USE_QT)
  qt_use_modules(VIVE.mpi Core)
ENDIF(COVISE_USE_QT)
COVISE_INSTALL_TARGET(VIVE.mpi)
